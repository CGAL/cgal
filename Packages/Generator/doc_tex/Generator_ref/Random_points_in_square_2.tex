\begin{ccRefClass}{Random_points_in_square_2<Point_2, Creator>}
\ccDefinition

The class \ccRefName\ is an input iterator creating points uniformly
distributed in a half-open square. The default \ccc{Creator} is
\ccc{Creator_uniform_2<Kernel_traits<P>::Kernel::RT,P>}.

\ccCreationVariable{g}
\ccConstructor{Random_points_in_square_2( double a, Random& rnd =
  default_random);}{%
  $g$ is an input iterator creating points of type \ccc{Point_2} uniformly
  distributed in the half-open square with side length $2 a$, centered
  at the origin, i.e.~$\forall p = \ccc{*g}:  -a \le p.x() < a$ and 
  $-a \le p.y() < a$~. 
  Two random numbers are needed from \ccc{rnd} for each point.
}
