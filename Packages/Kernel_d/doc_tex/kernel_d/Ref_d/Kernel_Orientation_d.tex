\begin{ccRefFunctionObjectConcept}{Kernel::Orientation_d}
A model for this must provide:

\ccCreationVariable{fo}

\ccMemberFunction{template <class ForwardIterator> 
Orientation operator()(ForwardIterator first, ForwardIterator last);}
{determines the orientation of the points of the tuple
 \ccc{A = tuple [first,last)} where $A$ consists of $d + 1$ points in
 $d$-space. This is the sign of the determinant
  \[ \left| \begin{array}{cccc}
  1 & 1 & 1 & 1 \\
  A[0] & A[1] & \dots & A[d]
  \end{array}  \right|  \]
  where \ccc{A[i]} denotes the cartesian coordinate vector of
  the $i$-th point in $A$.
  \ccPrecond \ccc{size [first,last) == d+1} and
  \ccc{A[i].dimension() == d} $\forall 0 \leq i \leq d$.
  \ccRequire The value type of \ccc{ForwardIterator} is
  \ccc{Kernel::Point_d}.  }

\end{ccRefFunctionObjectConcept}
