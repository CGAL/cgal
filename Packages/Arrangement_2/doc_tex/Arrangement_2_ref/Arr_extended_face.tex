\begin{ccRefClass}{Arr_extended_face<FaceBase,FData>}

\ccThree{FData &}{f.set_data(const FData & data)~}{}
\ccThreeToTwo

\ccDefinition
%============
  The \ccRefName\ class-template extends the face topological-features of the
  \dcel. It is parameterized by a face base-type \ccc{FaceBase} and a data type
  \ccc{FData} used to extend the face base-type.

\ccInclude{CGAL/Arr_extended_dcel.h}

\ccIsModel
%============
  \ccc{ArrangementDcelFace}

\ccInheritsFrom
%============
  \ccc{FaceBase}

\ccCreation
\ccCreationVariable{f}
%==========

  {assigns \ccVar{} with the contents of the \ccc{other} vertex.}

\ccAccessFunctions
%=================

\def\ccLongFuncLayout{\ccTrue}
  \ccMethod{const FData & data () const;}

   to a mutable data object is also available).}

\ccModifiers
%============
