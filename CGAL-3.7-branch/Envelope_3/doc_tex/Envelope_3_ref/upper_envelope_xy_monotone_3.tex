\begin{ccRefFunction}{upper_envelope_xy_monotone_3}

\ccInclude{CGAL/envelope_3.h}

\ccFunction{template<class InputIterator, class Traits>
            void upper_envelope_xy_monotone_3 (InputIterator begin, InputIterator end,
                                   Envelope_diagram_2<Traits>& diag);}
   {Computes the upper envelope of a set of $xy$-monotone surfaces in 
     $\mathbb{R}^3$, as given by the range \ccc{[begin, end)}. The lower 
     envelope is represented using the output maximization diagram \ccc{diag}.
    \ccPrecond{The value-type of \ccc{InputIterator} is
               \ccc{Traits::Xy_monotone_surface_3}.}}
