\begin{ccRefClass}{Skin_surface_sphere_3<Kernel>}
\label{refSkinsurfacesphere}
\ccDefinition
  
The class \ccRefName\ represents a sphere specifically designed for
the skin surface package.

\ccInclude{CGAL/Skin_surface_quadratic_surface_3.h}

\ccParameters 
%
It is parameterized by a kernel that must provide types for points,
vectors, segments and weighted points.

\ccInheritsFrom \ccc{Skin_surface_quadratic_surface_3<Kernel>}

\ccTypes
The geometric types used by the sphere:
\ccThree{typedef Weighted_point<Point, RT> x}{Triangulated}{}
\ccThreeToTwo

\ccTypedef{typedef Kernel::RT                  RT;}{}
\ccGlue
\ccTypedef{typedef Kernel::Point_3             Point;}{}
\ccGlue
\ccTypedef{typedef Kernel::Vector_3            Vector;}{}
\ccGlue
\ccTypedef{typedef Kernel::Segment_3           Segment;}{}
\ccGlue
\ccTypedef{typedef Weighted_point<Point, RT> Weighted_point;}{}

\ccCreation
\ccCreationVariable{surf}  %% choose variable name
\ccThree{Point }{Point}{}
\ccThreeToTwo

\ccConstructor{Skin_surface_sphere_3(Weighted_point wp, RT s, int
  orient);}{The center of the sphere is defined by the location of the
  weighted point and the radius is defined as $\sqrt{s P}$, where $P$
  is the weight of \ccc{wp}. The orientation is -1 or 1 and }

\ccOperations
\ccThree{virtual Vector }{ to_surfaceconst Point p0, const Point }{}
\ccThreeToTwo

\ccMethod{Point to_surface(const Point &p0, const Point &p1);}
{Returns a point close to the intersection point of the line segment
  \ccc{p0p1} with the surface.}
%
\ccGlue 
%
\ccMethod{Point to_surface(const Segment &s);}{Equal to
  \ccc{to_surface(s.source(), s.target())}.}

The following two functions give an implicit representation of the
skin surface. Each quadratic surface is defined as the zero set of a
scalar valued function. Using the mixed complex, we define the skin
surface as a piecewise implicit function, and this function is $C^1$.

\ccMethod{virtual RT value(const Point &p) const = 0;}{Returns the
  function value of the quadric.}

\ccGlue \ccMethod{virtual Vector gradient(const Point &p) = 0;
}{Returns the gradient in \ccc{p}. }
