\ccDefinition
  
The class \ccRefName\ is a model of \ccc{MeshCellCriteria_3}. It provides
uniform size and/or shape criteria for the  mesh tetrahedra.

\ccInclude{CGAL/Mesh_cell_criteria_3.h}

\ccParameters
The parameter \ccc{Tr} must be identical to the nested type
\ccc{Triangulation} of the instance used as model of
\ccc{MeshComplex_3InTriangulation_3}.

\ccIsModel
\ccc{MeshCellCriteria_3}


\ccTypes

\ccTypedef{typedef Tr::FT FT;}{Numerical type}

\ccCreation
\ccCreationVariable{mcc}  %% choose variable name

\ccConstructor{Mesh_cell_criteria_3(
  FT radius_edge_bound,
  FT radius_bound);}
{Returns an object to serve as default criteria  for cells. The argument 
\ccc{radius_edge_bound} is the  upper bound for the radius-edge ratio
of the tetrahedra. The argument \ccc{radius_bound} is a uniform upper bound
for the circumradii of the tetrahedra in the mesh. See
section~\ref{introsec:param} for further details.
Note that if one parameter is set to 0, then its corresponding criteria is ignored.}
