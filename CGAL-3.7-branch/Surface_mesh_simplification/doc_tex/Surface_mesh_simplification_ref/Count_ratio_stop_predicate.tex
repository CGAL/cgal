\ccDefinition

The class \ccRefName\ provides a model for the \ccc{StopPredicate} concept.
It has one template argument: the type of surface being simplified,
which must be a model of the \ccc{EdgeCollapsableMesh} concept.
It returns \ccc{true} when the relation between the initial and current number 
of edges drops below a certain ratio.

\ccInclude{CGAL/Surface_mesh_simplification/Policies/Edge_collapse/Count_ratio_stop_predicate.h}

\ccIsModel
\ccc{StopPredicate}

\ccCreation
\ccCreationVariable{sp}  %% choose variable name

\ccConstructor{Count_ratio_stop_predicate<ECM>( double ratio );}
{Initializes the predicate establishing the \ccc{ratio}.} 

\ccOperations
\ccTagFullDeclarations
\ccCreationVariable{sp}  %% choose variable name

  \ccMemberFunction{bool operator()( FT const&      current_cost
                    , Profile const& edge_profile
                    , size_type      initial_count
                    , size_type      current_count
                    ) const ;}
  {Returns $( ((double)current_count / (double)initial_count) < ratio)$. 
  All other parameters are ignored (but exist since this is a generic policy).}

\ccSeeAlso
\ccc{CGAL::Surface_mesh_simplification::Count_stop_predicate<ECM>}
