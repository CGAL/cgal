\ccHeading{Operators for Output}

The output operator is defined for all geometric classes in the \cgal\ kernel.

\ccFunction{template<class R> Qt_widget& operator<<(Qt_widget& win, const
Point_2<R>& p);}{}

\ccFunction{template<class R> Qt_widget& operator<<(Qt_widget& win, const
Segment_2<R>& s);}{}

\ccFunction{template<class R> Qt_widget& operator<<(Qt_widget& win, const
Line_2<R>& l);}{}

\ccFunction{template<class R> Qt_widget& operator<<(Qt_widget& win, const
Ray_2<R>& r);}{}

\ccFunction{template<class R> Qt_widget& operator<<(Qt_widget& win, const
Triangle_2<R>& t);}{}

\ccFunction{template<class R>Qt_widget& operator<<(Qt_widget& win, const 
Iso_rectangle_2<R>& r);}{}
