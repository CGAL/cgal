\begin{ccRefFunction}{subdivide_skin_surface_mesh_3<SkinSurface_3, Polyhedron_3>}
  \ccDefinition

  The function \ccRefName\ subdivides a skin surface mesh constructed
  by the function \ccc{mesh_skin_surface_3<SkinSurface_3, Polyhedron_3>}.

  \ccInclude{CGAL/subdivide_skin_surface_mesh_3.h}

  \ccThree{void}{a}{}
  \ccThreeToTwo

  \ccFunction{void
    subdivide_skin_surface_mesh_3<SkinSurface_3,Polyhedron_3> (const
    SkinSurface_3 &skin_surface, Polyhedron_3 &p, int nSubdiv =
    1);}{Subdivides the skin surface using \ccc{nSubdiv} 1-4 split
    operations (each triangle is split into four sub-triangles) and
    the new vertices are moved towards the skin surface.}

\end{ccRefFunction}
