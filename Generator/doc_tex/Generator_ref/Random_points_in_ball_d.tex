distributed in an open ball in any dimension.

\ccCreationVariable{g}
\ccConstructor{Random_points_in_ball_d(int dim, double r);}{%
  $g$ is an input iterator creating points of type \ccc{Point_d} uniformly
  distributed in the open ball in dimension $dim$ with radius $r$,
  i.e.~$|\ccc{*g}| < r$~.  $2\cdot dim+1$ random numbers are needed from
  \ccc{rnd} for each point.
