cmake_minimum_required(VERSION 3.12...3.31)

project( Variational_medial_axis_Tests )

find_package(CGAL REQUIRED)
find_package(Eigen3 REQUIRED 3.2.0)
find_package(TBB QUIET)
include(CGAL_Eigen3_support)
include(CGAL_TBB_support)
if(TARGET CGAL::Eigen3_support)
  create_single_source_cgal_program("variational_medial_axis_sampling_test.cpp")
  target_link_libraries(variational_medial_axis_sampling_test PRIVATE CGAL::Eigen3_support)
  if(TARGET CGAL::TBB_support)
    message(STATUS "Found TBB - enabling parallel support")
    target_link_libraries(variational_medial_axis_sampling_test PRIVATE CGAL::TBB_support)
    target_compile_definitions(variational_medial_axis_sampling_test PRIVATE CGAL_LINKED_WITH_TBB)
  else()
    message(STATUS "TBB not found - parallel support disabled")
  endif()
   create_single_source_cgal_program("fast_winding_number_test.cpp")
  target_link_libraries(fast_winding_number_test PRIVATE CGAL::Eigen3_support)
else()
  message("NOTICE: These tests require the Eigen library (3.2 or greater), and will not be compiled.")
endif()
