\begin{ccPkgDescription}{Approximation of Ridges and Umbilics on
    Triangulated Surface Meshes \label{Pkg:Ridges_3}}
  \ccPkgHowToCiteCgal{cgal:cp-arutsm-12} 
%
  \ccPkgSummary{Global features related to curvature extrema encode
    important informations used in segmentation, registration,
    matching and surface analysis. Given pointwise estimations of
    local differential quantities, this package allows the
    approximation of differential features on a triangulated surface
    mesh. Such curvature related features are curves: ridges or
    crests, and points: umbilics.}
%
  \ccPkgDependsOn{Solvers as \ccThirdPartyEigen, or \ccThirdPartyLapack\ and \ccThirdPartyBlas} 
%
  \ccPkgIntroducedInCGAL{3.3} \ccPkgLicense{\ccLicenseGPL}
